class Solution {
public:
    int maxArea(vector<int>& height) {
        if(height.empty()) return 0;
        int left = 0;
        int right = height.size()-1;
        
        int res = 0;
        while(left < right)
        {  
            height[left] > height[right]?res=max(res,(right-left)*height[right--]):
                                         res = max(res, (right-left)*height[left++]);
            // if(height[left] > height[right])
            // {

            //     right--;
            // }
            // else
            // {
            //     res = max(res,(right-left)*height[left]);
            //     left++;
            // }
        }

        return res;
    }
};
